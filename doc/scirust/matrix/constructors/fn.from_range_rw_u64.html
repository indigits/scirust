pub fn from_range_rw_u64(rows: usize, cols: usize, start: u64, stop: u64) -> MatrixU64

Returns an 64-bit unsigned integer matrix whose entries are picked up from a range in row wise order.

See from_range_rw function for further discussion.

Examples

use scirust::api::{from_range_rw_u64, Shape};

let m = from_range_rw_u64(4, 4, 0, 16);
for i in 0..16{
    let r = i >> 2;
    let c = i & 3;
    assert_eq!(m.get(r, c).unwrap(), i as u64);
}
